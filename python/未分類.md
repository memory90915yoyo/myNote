### code golf ###

`assign`

	# beginner python
	a = 0
	b = 0
	c = 0

	# proper python
	a, b, c = 0, 0, 0

	# code golf
	a = b = c = 0

	# beginner python
	a = 'a'
	b = 'b'
	c = 'c'

	# proper python
	a, b, c = 'a', 'b', 'c'

	# code golf
	a, b, c= 'abc'


`if else`

	# beginner python
	if a < 0:
	    b = 2 * a
	else:
	    b = 3 * a

	# proper python
	b = 2 * a if a < 0 else 3 * a

	# code golf 1
	b = a < 0 and 2 * a or 3 * a
	# code golf 2 
	b = a * (3, 2)[a<0]


`list`

	# reverse list, [1, 2, 3, 4] => [4, 3, 2, 1]
	[1, 2, 3, 4][::-1]

	#  combine two list, [a, b, c, d] + [1, 2, 3, 4] => [a1, b2, c3, d4]
	a = list("abcd")
	b = list("1234")

	[x+y for x, y in zip(a,b)]
	map(lambda x, y: x+y, a, b)
	map("".join, zip(a, b))
	map(str.__add__, a, b)
